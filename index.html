    function formatRelativeTime(ts) {

      const diff = Date.now() - ts;

      const m = Math.round(diff / 60000);

      if (m < 1) return "Just now";

      if (m < 60) return `${m} min ago`;

      const h = Math.round(m / 60);

      if (h < 24) return `${h} h ago`;

      const d = Math.round(h / 24);

      return `${d} d ago`;

    }

      if (p.status !== "active") return;

      const total = p.lockedVotingPower || 0;

      if (total <= 0) return;

      const yesFrac = p.yesPower / total;

      const noFrac = p.noPower / total;

      if (yesFrac >= threshold) {

          target.balance -= p.amount;

          id: "hist-" + nowTs + "-" + p.id + "-approved",

    function timeRemainingLabel(p) {

      if (p.status !== "active") return "Completed";

      const now = Date.now();

      const remaining = p.expiresAt - now;

      if (remaining <= 0) return "Expired";

      const hours = Math.floor(remaining / (60 * 60 * 1000));

      const minutes = Math.floor((remaining % (60 * 60 * 1000)) / (60 * 1000));

      if (hours <= 0 && minutes <= 0) return "⏱ Final seconds";

      if (hours <= 0) return `⏱ ${minutes} min left`;

      return `⏱ ${hours} h ${minutes} min left`;

    }

    function castVote(proposalId, memberId, choice) {

      const p = proposals.find(p => p.id === proposalId);

      if (!p || p.status !== "active") return;

      const member = getMember(memberId);

      const power = getVotingPowerForBalance(member.balance);

      if (member.balance < 200 || power <= 0) {

        alert("You currently cannot vote. Members need at least 200 AURA to vote.");

        return;

      }

      if (choice === "yes") {

        if (p.yesVotes[memberId]) {

          // Already yes; do nothing

        } else if (p.noVotes[memberId]) {

          p.noPower -= power;

          delete p.noVotes[memberId];

          p.yesVotes[memberId] = true;

          p.yesPower += power;

        } else {

          p.yesVotes[memberId] = true;

          p.yesPower += power;

        }

      } else if (choice === "no") {

        if (p.noVotes[memberId]) {

          // Already no

        } else if (p.yesVotes[memberId]) {

          p.yesPower -= power;

          delete p.yesVotes[memberId];

          p.noVotes[memberId] = true;

          p.noPower += power;

        } else {

          p.noVotes[memberId] = true;

          p.noPower += power;

        }

      }

      updateProposalStatus(p);

      renderAll();

    }


    function handleCreateProposal(event) {

      event.preventDefault();

      const currentUser = getCurrentUser();

      const hasActive = proposals.some(p => p.proposerId === currentUser.id && p.status === "active");

      if (hasActive) {

        alert("You already have an active proposal. Wait until it resolves or expires.");

        return;

      }

      if (currentUser.balance < 100) {

        alert("You need at least 100 AURA to stake for a proposal.");

        return;

      }

      const targetId = document.getElementById("targetMemberSelect").value;

      const type = document.getElementById("proposalTypeSelect").value;

      let amount = parseInt(document.getElementById("proposalAmountInput").value || "0", 10);

      if (isNaN(amount)) amount = 0;

      if (amount < 1 || amount > 500) {

        alert("Proposal amount must be between 1 and 500 AURA.");

        return;

      }

      const title = document.getElementById("proposalTitleInput").value.trim();

      const description = document.getElementById("proposalDescInput").value.trim();

      if (!title || !description) {

        alert("Please fill title and description.");

        return;

      }

      const createdAt = Date.now();

      const newProposal = {

        id: "p-" + createdAt,

        title,

        description,

        proposerId: currentUser.id,

        targetId,

        type,

        amount,

        stakeAmount: 100,

        createdAt,

        expiresAt: createdAt + HOURS_12_MS,

        lockedVotingPower: 0,

        yesPower: 0,

        noPower: 0,

        yesVotes: {},

        noVotes: {},

        status: "active",

        outcome: null

      };

      let total = 0;

      members.forEach(m => {

        const power = getVotingPowerForBalance(m.balance);

        if (power > 0) total += power;

      });

      newProposal.lockedVotingPower = total;


      currentUser.balance -= 100;

      history.unshift({

        id: "hist-" + createdAt + "-stake",

        type: "stake-burned",

        amount: -100,

        memberId: currentUser.id,

        proposerId: currentUser.id,

        targetId,

        status: "Stake staked",

        detail: `Stake of 100 AURA locked for proposal: ${title}`,

        timestamp: createdAt

      });


      proposals.unshift(newProposal);


      document.getElementById("proposalAmountInput").value = "100";

      document.getElementById("proposalTitleInput").value = "";

      document.getElementById("proposalDescInput").value = "";

      updateProposalSummaryPreview();

      updateProposalCreatorStatus();

      renderAll();

      alert("Proposal created! Your 100 AURA stake is now locked in.");

    }
